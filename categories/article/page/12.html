#include&lt;iostream&gt;
#include&lt;cstdio&gt;
#include&lt;vector&gt;

vector&lt;int&gt;chirld[10002];
int dp[10002] = {0}
    ,degree[10002] = {0}
    ,isJudged[10002] = {0};

{
    int n,i,a,b;
    bool isnone = true;
    scanf(&#34;%d&#34;,&amp;n);
    for(i = 1 ; i &lt; n ; i ++)
    {
        scanf(&#34;%d %d&#34;,&amp;a,&amp;b);
        chirld[a].push_back(b);
        chirld[b].push_back(a);
    }

    search(1 , n);

    for(i = 1 ; i &lt;= n ; i ++)
        if(dp[i] * 2 &lt;= n)
            printf(&#34;%d\n&#34;,i) , isnone = false;

}

int search(int pos,int &amp;n)
{
    int i,j;
    degree[pos] = 1;
    isJudged[pos] = 1;
    dp[pos] = 0;

    for(i = 0 ; i &lt; chirld[pos].size() ; i ++)
    {
        if(!isJudged[chirld[pos].at(i)])//еҰӮжһңе·Із»ҸеҲӨж–ӯиҝҮе°ұжҳҜзҲ¶дәІиҠӮзӮ№дәҶ

            if(dp[pos] &lt; degree[chirld[pos].at(i)])
                dp[pos] = degree[chirld[pos].at(i)];
        }
    }

    if(dp[pos] &lt; n - degree[pos])//еҲӨж–ӯзҲ¶дәІиҠӮзӮ№ж•°йҮҸ
        dp[pos] = n - degree[pos];
    return degree[pos];
}

//жңҖй•ҝеҚ•и°ғеӯҗеәҸеҲ— еӨҚжқӮеәҰnlog(n)
//еҸӮж•°(еҺҹеәҸеҲ—,еәҸеҲ—й•ҝеәҰ,з”ҹжҲҗзҡ„еәҸеҲ—),дј е…ҘеәҸеҲ—й•ҝеәҰеҝ…йЎ»еӨ§дәҺ0

//LIScmpдёәе…ізі»еҮҪж•°,еҺҹеҮҪж•°иЎЁжҳҺlengthRecordдёәйҖ’еўһ(дёҚеҗ«зӯүдәҺ)
typedef double LISTYPE;
#define LISMAXN 10000
int LIScmp(LISTYPE a,LISTYPE b)
{
    return a &lt; b;
}
long LISLength(LISTYPE list[],long n,LISTYPE lengthRecord[])
{
    long length = 1,lth;
    LISTYPE lR[LISMAXN];
    lR[0] = list[0];

    for(int i = 1 ; i &lt; n ; i ++)
    {
        //дәҢеҲҶжҹҘжүҫ,еӨҚжқӮеәҰ log(n)
        int b,e,m;
        b = 0;
        e = length - 1;
        while(b &lt;= e &amp;&amp; e &gt;= 0)
        {
            m = (b + e) / 2;
            if(LIScmp(lR[m],list[i]))
                b = m + 1;
            else
                e = m - 1;
        }
        lR[b] = list[i];
        if(b &gt;= length)
            length ++;
    }
    /*
    *и®Ўз®—еәҸеҲ—йғЁеҲҶ
    *еӨҚжқӮеәҰnlog(n)
    */
    lth = 1;
    for(int i = 1 ; i &lt; n ; i ++)
    {
        //дәҢеҲҶжҹҘжүҫ,еӨҚжқӮеәҰ log(n)
        int b,e,m;
        b = 0;
        e = lth - 1;
        while(b &lt;= e &amp;&amp; e &gt;= 0)
        {
            m = (b + e) / 2;
            if(LIScmp(lR[m],list[i]))
                b = m + 1;
            else
                e = m - 1;
        }
        lR[b] = list[i];
        if(b &gt;= lth)
            lth ++;
        if(lth == length)
        {
            for(b = 0 ; b &lt; length ; b ++)
                lengthRecord[b] = lR[b];
            break;
        }
    }
    //и®Ўз®—еәҸеҲ—йғЁеҲҶд»Јз ҒдёҺд№ӢеүҚзҡ„зұ»дјј,еҸҜд»ҘзӣҙжҺҘCopyз„¶еҗҺдҝ®ж”№
    return length;
}

PrimeType GetPrimeLength(const long n)
{
    PrimeType tmpLen = MAXNUM;
    long tmpPos = 0,left = n - 1;
    PrimeType sumLen = 0;

    for(long i = 1 ; i &lt; n ; i ++)
        PrimeRecord[i] = dis[0][i];
    while(left --)
    {
        tmpLen = MAXNUM;
        for(long i = 1 ; i &lt; n ; i ++)
            if(!isLined[i] &amp;&amp; PrimeRecord[i] &lt; tmpLen)
                tmpPos = i,tmpLen = PrimeRecord[i];

        sumLen += tmpLen;
        isLined[tmpPos] ++;
        for(long i = 1 ; i &lt; n ; i ++)
            if(dis[tmpPos][i] &lt; PrimeRecord[i])
                PrimeRecord[i] = dis[tmpPos][i];
    }

    return sumLen;
}

{

    {
        x0 = x;
