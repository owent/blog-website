A Grey Area

A题是很诡异的统计，是一道纯模拟就能过的题，其他的不多说了

代码：

#include&lt;iostream&gt;
using namespace std;
int pe[20];
int main()
{
    int n,w;
    while(cin&gt;&gt;n&gt;&gt;w, n || w)
    {
        int value;
        int peNum = 0;
        int Max = 0;
        int valueMax = 0;
        memset(pe,0,sizeof(pe));
        while(n --)
        {
            scanf(&#34;%d&#34;,&amp;value);
            while(value &gt;= Max)
            {
                peNum ++;
                Max = peNum * w;
            }
            pe[value / w] ++;
            if(pe[value / w] &gt; valueMax)
                valueMax = pe[value / w];
        }

        double output = 0.01;
        for(int i = 0 ; i &lt; peNum ; i ++)
            output += ((double)pe[i] / (double)valueMax)
                * ((double)(peNum - i - 1) / (double)(peNum - 1));

        printf(&#34;%lf\n&#34;,output);
    }
    return 0;
}

B Expected Allowance

B题和A题类似，也是纯暴力

代码：

#include&lt;iostream&gt;
#include&lt;cmath&gt;
using namespace std;
long value[10000];

void cal(long now,int pos,int n,int m)
{
    for(int i = 1 ; i &lt;= m ; i ++)
        if(pos == n)
            value[now + i] ++;
        else
            cal(now + i,pos + 1,n,m);
}


int main()
{
    long m,n,k;
    while(scanf(&#34;%ld %ld %ld&#34;,&amp;n,&amp;m,&amp;k), n || m || k )
    {
        memset(value,0,sizeof(value));
        cal(0,1,n,m);
        double output = 0.0;
        double fm = pow((double)m,(double)n);
        for(long i = n ; i &lt;= n * m ; i ++)
            if(i - k &lt;= 0)
                output += (double)value[i] / fm;
            else
                output += (double)value[i] / fm * (double)(i - k);

        printf(&#34;%.8lf\n&#34;,output);
    }
    return 0;
}

第七场

B要注意一下判断的优先级.

代码如下：

#include&lt;iostream&gt;
#include&lt;cstring&gt;
using namespace std;
double bank[10][10000];
char cmd[20];
int isExist[10][10000];
int main()
{
    int n = 1;
    //freopen(&#34;b.in&#34;,&#34;r&#34;,stdin);
    while(scanf(&#34;%d&#34;,&amp;n),n)
    {
        memset(bank,0,sizeof(bank));
        memset(isExist,0,sizeof(isExist));
        while(n --)
        {
            int tmpInt1,tmpInt2;
            double tmpMoney;
            scanf(&#34;%d/%d %lf&#34;,&amp;tmpInt1,&amp;tmpInt2,&amp;tmpMoney);
            isExist[tmpInt2][tmpInt1] = 1;
            bank[tmpInt2][tmpInt1] = tmpMoney;
        }
        while(scanf(&#34;%s&#34;,cmd) , strcmp(cmd,&#34;end&#34;))
        {
            if(!strcmp(cmd,&#34;withdraw&#34;))
            {
                int tmpInt1,tmpInt2;
                double tmpMoney;
                scanf(&#34;%d/%d %lf&#34;,&amp;tmpInt1,&amp;tmpInt2,&amp;tmpMoney);
                printf(&#34;withdraw %.2lf: &#34;,tmpMoney);
                if(isExist[tmpInt2][tmpInt1])
                {
                    if(bank[tmpInt2][tmpInt1] &lt; tmpMoney)
                        printf(&#34;insufficient funds\n&#34;);
                    else
                    {
                        printf(&#34;ok\n&#34;);
                        bank[tmpInt2][tmpInt1] -= tmpMoney;
                    }
                }
                else
                    printf(&#34;no such account\n&#34;);
            }
            else if(!strcmp(cmd,&#34;deposit&#34;))
            {
                int tmpInt1,tmpInt2;
                double tmpMoney;
                scanf(&#34;%d/%d %lf&#34;,&amp;tmpInt1,&amp;tmpInt2,&amp;tmpMoney);
                printf(&#34;deposit %.2lf: &#34;,tmpMoney);
                if(isExist[tmpInt2][tmpInt1])
                {
                    printf(&#34;ok\n&#34;);
                    bank[tmpInt2][tmpInt1] += tmpMoney;
                }
                else
                    printf(&#34;no such account\n&#34;);
            }
            else if(!strcmp(cmd,&#34;create&#34;))
            {
                int tmpInt1,tmpInt2;
                scanf(&#34;%d/%d&#34;,&amp;tmpInt1,&amp;tmpInt2);
                printf(&#34;create: &#34;);
                if(isExist[tmpInt2][tmpInt1])
                    printf(&#34;already exists\n&#34;);
                else
                {
                    printf(&#34;ok\n&#34;);
                    isExist[tmpInt2][tmpInt1] = 1;
                    bank[tmpInt2][tmpInt1] = 0;
                }
            }
            else if(!strcmp(cmd,&#34;transfer&#34;))
            {
                int tmpInt1,tmpInt2,tmpInt3,tmpInt4;
                double tmpMoney;
                scanf(&#34;%d/%d %d/%d %lf&#34;,&amp;tmpInt1,&amp;tmpInt2,&amp;tmpInt3,&amp;tmpInt4,&amp;tmpMoney);
                printf(&#34;transfer %.2lf: &#34;,tmpMoney);
                if(isExist[tmpInt2][tmpInt1] &amp;&amp; isExist[tmpInt4][tmpInt3])
                {
                    if(tmpInt2 == tmpInt4 &amp;&amp; tmpInt1 == tmpInt3)
                        printf(&#34;same account\n&#34;);
                    else if(bank[tmpInt2][tmpInt1] &lt; tmpMoney)
                        printf(&#34;insufficient funds\n&#34;);
                    else
                    {
                        if(tmpInt2 == tmpInt4)
                            printf(&#34;ok\n&#34;);
                        else
                            printf(&#34;interbank\n&#34;);
                        bank[tmpInt2][tmpInt1] -= tmpMoney;
                        bank[tmpInt4][tmpInt3] += tmpMoney;
                    }
                }
                else
                    printf(&#34;no such account\n&#34;);
            }
        }
        printf(&#34;end\n\n&#34;);
    }
    printf(&#34;goodbye&#34;);
    return 0;
}

#include&lt;iostream&gt;
#include&lt;cstring&gt;
using namespace std;
char Issuer[12];
struct Member
{
    char name[30];
    double money;
    char type[10];
};
Member mem[1001];
int main()
{
    int n;
    //freopen(&#34;e.txt&#34;,&#34;r&#34;,stdin);
    while(scanf(&#34;%d %s&#34;,&amp;n,Issuer), strcmp(Issuer,&#34;END&#34;) || n)
    {
        int i,j;
        for(i = 0 ; i &lt; n ; i ++)
            scanf(&#34;%s %s %lf&#34;,mem[i].name,mem[i].type,&amp;mem[i].money);

        printf(&#34;%s\n&#34;,Issuer);
        int num = 0;
        for(i = 0 ; i &lt; n ; i ++)
        {
            num = 0;
            printf(&#34;%s:&#34;,mem[i].name);
            for(j = 0 ; j &lt; n ; j ++)
            {
                if(mem[i].type[0] == mem[j].type[0])
                    continue;
                if(mem[i].type[0] == &#39;b&#39;
                    &amp;&amp; mem[j].money &lt;= mem[i].money)
                    printf(&#34; %s&#34;,mem[j].name),num ++;
                else if(mem[i].type[0] == &#39;s&#39;
                    &amp;&amp; mem[j].money &gt;= mem[i].money)
                    printf(&#34; %s&#34;,mem[j].name),num ++;
            }
            if(num)
                putchar(&#39;\n&#39;);
            else
                printf(&#34; NO-ONE\n&#34;);
        }
    }
    return 0;
}
