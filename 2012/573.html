JSON显示库 -- showJson （Javascript）


现在的web的js开发很方便啊，但是碰到iframe里的东西还是不方便看到变量的内容，所以就写了这么个看json内容的玩意，还可以当控制台输出用。

很简单,有需要以后开发新功能

/** 
 * Show json in a new page.(For debug)
 *
 * Licensed under the MIT or GPL Version 3 licenses.

 * @link http://www.owent.net
 * @history 
 * 		2012.12.27 
 * 			1. 改进JSON内对象循环引用时的导致的栈溢出问题
 * 			2. 引入层级路径
 * 			3. 引入锚点
 */
 
 /**
  * Object转为字符串
  * @param {Object} json Object对象 
  * @param {Number} tab 缩进数量
  * @param {Object} rules 应用规则
  * @param {String} path 当前路径
  * @return 生成的HTML
  */
function json2String(json, tab, rules, path) {
    var tabStr = &#34;&#34;, singleTab = &#34;&amp;nbsp;&amp;nbsp;&amp;nbsp;&amp;nbsp;&#34;, isClear = false;
    var txt = &#34;&#34;;
    
    if (!rules) {
        rules = {
            id: &#34;__json2string_process&#34; + (new Date).getTime().toString() + (json2String.baseIndex ++),
            objs: []
        };
        isClear = true;
    }
    
    tab = tab || 1;
    if (!path) {
        path = &#34;{ROOT}&#34;;
        txt += &#34;&lt;a name=\&#34;&#34; + rules.id + &#34;.&#34; + path + &#34;\&#34;&gt; &lt;/a&gt;&#34;;
    }
    
    for (var i = 0; i &lt; tab; i ++) {
        tabStr += singleTab;
    }
    
    if (!json &amp;&amp; json !== 0) {
        txt += &#39;&lt;span style=&#34;color: Red;&#34;&gt;null&lt;/span&gt;&#39;;
    } else if (typeof(json) == &#34;object&#34;) {
        var prefix = &#39;&lt;span style=&#34;color: DarkGray; font-weight: bolder;&#34;&gt;{&lt;/span&gt;&#39;, suffix = &#39;&lt;span style=&#34;color: DarkGray; font-weight: bolder;&#34;&gt;}&lt;/span&gt;&#39;;
        try {		
            if (toString.call(json).match(/array/i)) {
                prefix = &#39;&lt;span style=&#34;color: Orange;&#34;&gt;[&lt;/span&gt;&#39;;
                suffix = &#39;&lt;span style=&#34;color: Orange;&#34;&gt;]&lt;/span&gt;&#39;;
            }
        } catch(e) {
            // hoho ...
        }
        
        if (json[rules.id]) {
            txt += prefix + &#34;此处引用 &lt;a style=\&#34;text-decoration: underline; color: Blue;\&#34; href=\&#34;#&#34; + rules.id + &#34;.&#34; + json[rules.id] + &#34;\&#34; target=\&#34;_self\&#34; title=\&#34;跳转到引用 &#34; + json[rules.id] + &#34;\&#34; &gt;&#34; + 
                json[rules.id] + &#34;&lt;/a&gt;&#34; + suffix;
            return txt;
        }
        json[rules.id] = path;
        rules.objs.push(json);
        
        txt += prefix + &#34;&lt;br /&gt;\r\n&#34;;
        var first = true;
        for (var key in json) {
            if (key == rules.id)
                continue;
                
            if (first)
                first = false;
            else
                txt += &#34;,&lt;br /&gt;\r\n&#34;;
            txt += tabStr + &#34;&lt;a name=\&#34;&#34; + rules.id + &#34;.&#34; + path + &#34;.&#34; + key + &#34;\&#34;&gt; &lt;/a&gt;&#34; + &#34;&lt;strong&gt;&#34; + 
                key + &#34;&lt;/strong&gt; : &#34; +	json2String(json[key], tab + 1, rules, path + &#34;.&#34; + key);
        }
        txt += &#34;&lt;br /&gt;\r\n&#34; + tabStr.substr(0, tabStr.length - singleTab.length) + suffix;
    } else if (typeof(json) == &#34;string&#34;) {
        txt += &#39;&lt;span style=&#34;color: Green;&#34;&gt;&#34;&#39; + json + &#39;&#34;&lt;/span&gt;&#39;;
    } else if (typeof(json) == &#34;number&#34;) {
        txt += &#39;&lt;span style=&#34;color: Blue;&#34;&gt;&#39; + json + &#39;&lt;/span&gt;&#39;;
    } else if (typeof(json) == &#34;function&#34;) {
        txt += &#39;&lt;span style=&#34;color: Purple;&#34;&gt;&#39; + json + &#39;&lt;/span&gt;&#39;;
    } else {
        txt += json.toString();
    }
    
    if (isClear) {
        for (var i = 0; i &lt; rules.objs.length; ++ i) {
            delete rules.objs[i][rules.id];
        }
        rules.objs = [];
    }
    
    return txt;
}
 
function showJson(json, title, windowName, dlg_opt) {
    json2String.baseIndex = json2String.baseIndex || 0;
    
    title = title || &#34;Json viewer&#34;;
    windowName = windowName || &#34;result&#34; + (new Date()).toGMTString();
    if (window.jQuery &amp;&amp; jQuery.fn &amp;&amp; jQuery.fn.dialog) {
        $(&#34;&lt;div&gt;&lt;/div&gt;&#34;).append(json2String(json)).css({
            &#34;text-align&#34;: &#34;left&#34;
        }).dialog($.extend({
            &#34;modal&#34;: false,
            &#34;title&#34;: title,
            &#34;width&#34;: 640,
            &#34;height&#34;: 480,
            &#34;buttons&#34;: {
                &#34;关闭&#34;: function() {
                    $( this ).dialog( &#34;close&#34; );
                }
            }
        }, dlg_opt));
    } else {
        var wnd = window.open(&#34;about:blank&#34;, &#34;_blank&#34;, &#34;height=600, width=800, toolbar=no, menubar=no, resizable=yes, location=no, status=no&#34;, true);
        if (!wnd)
            alert(&#34;窗口被拦截！&#34;);
        wnd.document.write(&#34;&lt;!DOCTYPE html&gt;&lt;html&gt;&lt;head&gt;&lt;title&gt;&#34; + title + &#34;&lt;/title&gt;&lt;/head&gt;&lt;body&gt;&#34; + json2String(json) + &#34;&lt;/body&gt;&lt;/html&gt;&#34;);
    }
    return wnd;
}
  1. 更新对不同类型着色
  2. 如果载入了jQuery UI 则使用jQuery UI的Dialog打开，用于解决嵌套iframe时浏览器拦截问题
  3. 解决Object循环引用时栈溢出问题，同时增加引用的指向锚点
