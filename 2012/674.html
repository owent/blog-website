Boost.Spirit еҲқдҪ“йӘҢ

  • йҰ–е…ҲжқҘиҜ•з”ЁQiеә“пјҡ
  • жҺҘдёӢжқҘжҳҜKarmaеә“пјҡ
  • жңҖеҗҺжҳҜLexеә“пјҡ
  • еҗ„з§ҚSampleзҡ„иҝҗиЎҢз»“жһңеҰӮдёӢпјҡ

дҪҝз”Ёд»Јз Ғз”ҹжҲҗд»Јз ҒжҳҜдёҖд»¶еҚҒеҲҶзҫҺеҰҷзҡ„дәӢжғ…пјҢдәҺжҳҜжңүдәҶеҗ„з§Қд»Јз Ғз”ҹжҲҗеҷЁгҖӮдҪҶжҳҜз”ҹжҲҗд»Јз ҒпјҢж„Ҹе‘ізқҖиҰҒжңүеҜ№з”ҹжҲҗи§„еҲҷзҡ„еҲҶжһҗе’ҢеӨ„зҗҶгҖӮ Boost.Spirit е°ұжҳҜиҝҷд№ҲдёҖдёӘиҜӯжі•еҲҶжһҗе·Ҙе…·пјҢе®ғе®һзҺ°дәҶеҜ№дёҠдёӢж–Үж— е…іж–Үжі•зҡ„LLеҲҶжһҗгҖӮж”ҜжҢҒEBNF(жү©еұ•е·ҙз§‘ж–ҜиҢғејҸ)гҖӮ Boost.Spirit зҡ„дҪҝз”Ёзңҹзҡ„жҳҜжҠҠжЁЎжқҝеөҢеҘ—з”ЁеҲ°дәҶжһҒиҮҙгҖӮзЎ®е®һиҝҷд№ҲеҒҡйҖ жҲҗдәҶйқһеёёејәзҡ„жү©еұ•жҖ§пјҢз”ҹжҲҗзҡ„д»Јз Ғд№ҹйқһеёёй«ҳж•ҲпјҢдҪҶжҳҜеөҢеҘ—зҡ„еӨӘеӨҚжқӮдәҶпјҢеҜ№дәҺеҲқеӯҰиҖ…иҖҢиЁҖзңҹеҝғйҡҫзңӢжҮӮгҖӮ дҪ иғҪжғіиұЎеңЁеӯҰд№ йҳ¶ж®өдёҖдёӘдёҚжҳҜеӨӘжҳҺзҷҪзҡ„й”ҷиҜҜеҜјиҮҙзј–иҜ‘еҷЁжҠҘеҮәзҡ„еҮ еҚҒеұӮжЁЎжқҝеөҢеҘ—й”ҷиҜҜдҝЎжҒҜзҡ„ж„ҹеҸ—еҗ—пјҹиҖҢдё”пјҢиҝҷд№ҲеӨҚжқӮзҡ„жЁЎжқҝеөҢеҘ—иҝҳзӣҙжҺҘеҜјиҮҙдәҶзј–иҜ‘йҖҹеәҰзҡ„е·Ёж…ўж— жҜ”гҖӮ е…¶е®һеңЁд№ӢеүҚпјҢжҲ‘е·Із»ҸдҪҝз”ЁиҝҮSpiritзҡ„ClassicзүҲжң¬пјҢеҚі1.XзүҲжң¬пјҢдҪҶжҳҜиҝҮеӨҡзҡ„еӨҚеҲ¶ж“ҚдҪңи®©жҲ‘и§үеҫ—еҪ“ж—¶з”Ёеҫ—еҫҲдҪҺж•ҲпјҢиҝҳеҘҪеҲҶжһҗзҡ„еҶ…е®№е№¶дёҚеӨҚжқӮжүҖд»ҘжІЎгҖӮдҪ“зҺ°еҮәжқҘ иҝҷеӣһе°ұжқҘз ”з©¶дёӢеҠҹиғҪжӣҙејәеҠІзҡ„2.X зүҲжң¬гҖӮ

Boost.Spirit V2 еӨ§дҪ“дёҠеҲҶдёәдёүдёӘйғЁеҲҶпјҢQiгҖҒKarmaе’ҢLex

Qi еә“дё»иҰҒжҳҜи§„еҲҷз”ҹжҲҗе’Ңи§ЈжһҗеҷЁпјҢдҪҝз”Ёж–№ејҸзұ»дјје·ҙз§‘ж–ҜиҢғејҸ Karma еә“еҲҷжҳҜж јејҸеҢ–иҫ“еҮәе·Ҙе…· Lex еә“жҳҜзұ»дјјFlexзҡ„и§„еҲҷз”ҹжҲҗе·Ҙе…·пјҢдҪҝз”ЁжӯЈеҲҷиЎЁиҫҫејҸпјҢжҹҗдәӣж—¶еҖҷжҜ”зӣҙжҺҘдҪҝз”ЁQiжӣҙе®№жҳ“зңӢжҮӮдёҖдәӣ

жіЁпјҡжүҖжңүзӨәдҫӢзҡ„жңҖз»ҲиҝҗиЎҢз»“жһңйғҪж”ҫеңЁжңҖеҗҺ

йҰ–е…ҲжқҘиҜ•з”ЁQiеә“пјҡ

Qiеә“жҳҜд»Ҙи§ЈжһҗеҷЁParserдёәж ёеҝғзҡ„пјҢйҰ–е…ҲжҸҗдҫӣдәҶдёҖдәӣеҹәжң¬зҡ„и§ЈжһҗеҷЁпјҢжҜ”еҰӮж•ҙеһӢгҖҒеӯ—з¬ҰгҖҒжө®зӮ№ж•°зӯүзӯүгҖӮ е…·дҪ“еҶ…е®№еҸӮи§ҒBoost.Spiritзҡ„QiйғЁеҲҶзҡ„Qi Parsers з« иҠӮ

QiиҝҳеҢ…еҗ«еұһжҖ§зҡ„е®ҡд№үпјҢеҸӮи§ҒQiйғЁеҲҶзҡ„ Compound Attribute Rules з« иҠӮпјҢеұһжҖ§е®ҡд№үдё»иҰҒжҳҜжҸҸиҝ°дәҶдёҚеҗҢзҡ„иҜӯжі•и§„еҲҷй”ҒдҪҝз”Ёзҡ„ж•°жҚ®з»“жһ„пјҢеё®еҠ©жҲ‘д»¬еҲӨж–ӯж•°жҚ®иҪ¬еӮЁе’ҢиҜ»еҸ–зҡ„гҖӮиҝҷйҮҢйқўд№ҹжҸҸиҝ°дәҶQiзҡ„и§ЈжһҗеҷЁж”ҜжҢҒзҡ„ж“ҚдҪңз¬ҰгҖӮ дҪҝз”ЁеұһжҖ§е®ҡд№үиҜҙжҳҺдёӯзҡ„ж“ҚдҪңз¬ҰгҖҒqi::ruleе’ҢдёҠдёҖжқЎжҸҗеҲ°зҡ„еҹәжң¬и§ЈжһҗеҷЁпјҢеҸҜд»Ҙз»„жҲҗеӨҚжқӮең°ж»Ўи¶іжҲ‘д»¬йңҖжұӮзҡ„и§Јжһҗи§„еҲҷ

еҸҰеӨ–е°ұжҳҜQiзҡ„еҠЁдҪңеҷЁйғЁеҲҶдәҶпјҢи§ҒQiйғЁеҲҶзҡ„ Parser Semantic Actions з« иҠӮпјҢеҠЁдҪңеҷЁз”ЁдәҺеӨ„зҗҶеҢ№й…ҚзҺ©и§ЈжһҗеҷЁд№ӢеҗҺзҡ„ж“ҚдҪңгҖӮ з”іжҳҺеҪўејҸжҳҜ parser[f] е…¶еҠЁдҪңеҮҪж•°зҡ„еҪўејҸж”ҜжҢҒд»ҘдёӢж јејҸ:

// еҰӮжһңfжҳҜжҷ®йҖҡжҲ–йқҷжҖҒеҮҪж•°пјҢfеҝ…йЎ»ж»Ўи¶ід»ҘдёӢеҪўејҸд№ӢдёҖ
void f(Attrib const&);
void f(Attrib const&, Context&);
void f(Attrib const&, Context&, bool&);

// еҰӮжһңfжҳҜд»ҝеҮҪж•°пјҢfеҝ…йЎ»йҮҚиҪҪдёӢеҲ—ж“ҚдҪңз¬Ұд№ӢдёҖ
void operator()(Attrib const&, unused_type, unused_type) const;
void operator()(Attrib const&, Context&, unused_type) const;
void operator()(Attrib const&, Context&, bool&) const;

// д»ҘдёҠзҡ„AttribйғҪжҢҮзҡ„жҳҜеұһжҖ§еҷЁзұ»еһӢ

еҸҰеӨ–пјҢBoost.Spiritиҝҳе®һзҺ°дәҶдёҖдёӘPhoenixиҫ…еҠ©жЎҶжһ¶пјҢиҝҷжҳҜз”ЁдәҺз”ҹжҲҗеҜ№зұ»дјјLambdaиЎЁиҫҫејҸзҡ„ж”ҜжҢҒзҡ„д»Јз Ғзҡ„гҖӮеҸҜд»Ҙз®ҖеҢ–еҫҲеӨҡж“ҚдҪңпјҢеҸӘиҰҒеӨҚеҗҲBoost.Phoenixзҡ„дёҖдәӣи§„еҲҷ иҝҷйҮҢжңүдёӘз»јеҗҲSampleпјҡ

// й«ҳзә§иҮӘе®ҡд№үActionеҮҪж•°
struct SpiritQiStringAssign {
    mutable std::string& strAssigned;
    SpiritQiStringAssign(std::string& strAttached): strAssigned(strAttached){}

    void operator()(std::vector const &strVecTestOut, boost::spirit::qi::unused_type, boost::spirit::qi::unused_type) const {
        strAssigned.assign(strVecTestOut.data(), strVecTestOut.size());
    }
};

void test_spirit_qi() {
    using namespace boost::spirit;
    using boost::phoenix::at;
    using boost::phoenix::ref;
    using boost::phoenix::push_back;
    using boost::spirit::ascii::space;
    using boost::spirit::ascii::char_;
    using boost::spirit::eol;
    using std::string;

    // ============================================================
    // ============ Spirit QI componentпјҢ и§„еҲҷжһ„йҖ е·Ҙе…· ============
    // ============================================================
    string strTest = "123.6 45 ABC";
    double dTestOut = 0.0;
    std::vector stIntTestOut;
    int iTestOut;
    string strTestOut;
    std::vector strVecTestOut;

    bool res = qi::parse(strTest.begin(), strTest.end(), qi::double_, dTestOut);
    std::cout<< "Spirit.Qi    => Line "<< std::setw(4)<< __LINE__<< ": QI::Parse double "<< (res?"true": "false")<< std::endl;

    dTestOut = 1.0;
    /**
     * еӨҚеҗҲи§„еҲҷж—¶еҸҜдҪҝз”ЁеӨҡз§Қж“ҚдҪңз¬ҰпјҢеҢ…жӢ¬ >>гҖҒ>гҖҒ*гҖҒ+гҖҒ-гҖҒ!гҖҒ& зӯүзӯү
     *     е…ідәҺеӨҚеҗҲи§„еҲҷз»„еҗҲзҡ„ж“ҚдҪңз¬Ұж–ҮжЎЈеҸӮз…§ Boost.Spirit еә“ Qi йғЁеҲҶ Compound Attribute Rules з« иҠӮ
     * 
     * qi::rule еҸҜз”ЁдәҺе»әз«ӢеӨҚжқӮиҜӯжі•и§„еҲҷпјҲжіЁж„ҸиҰҒз¬ҰеҗҲEBNFиҢғејҸпјҢиҰҒж¶ҲйҷӨе·ҰйҖ’еҪ’пјү
     *     е…ідәҺqi::ruleзҡ„ж–ҮжЎЈеҸӮз…§ Boost.Spirit еә“ Qi йғЁеҲҶ Nonterminals з« иҠӮ
     * 
     * QiиЎҢдёәеұһжҖ§пјҲз”іжҳҺеҪўејҸ и§„еҲҷ[еҮҪж•°/д»ҝеҮҪж•°]пјүжҺҘеҸ—зҡ„еҮҪж•°зұ»еһӢдёәпјҡ
     *     еҮҪж•°пјҡ
     *         void f(Attrib const&);
     *         void f(Attrib const&, Context&);
     *         void f(Attrib const&, Context&, bool&);
     *     д»ҝеҮҪж•°пјҲд»ҘдёӢдёәNйҖү1пјүпјҡ
     *         е®ҡд№ү void operator()(Attrib const&, Context&, bool&) const;
     *         е®ҡд№ү void operator()(Attrib const&, Context&, unused_type) const;
     *         е®ҡд№ү void operator()(Attrib const&, unused_type, unused_type) const;
     * д№ҹеҸҜд»Ҙз”Ё Boost.Phoenix жһ„йҖ QiиЎҢдёәеұһжҖ§
     *     зӨәдҫӢдёӯеҲҶеҲ«йҮҮз”ЁдәҶPhoenixзҡ„refгҖҒassignе’Ңpush_backжқҘиҺ·еҸ–иҫ“е…Ҙж•°жҚ®
     */
    qi::rule stQiRule = 
        // з¬¬дёҖйғЁеҲҶпјҢеј•з”ЁеҢ…иЈ…+зӯүеҸ·ж“ҚдҪңз¬ҰпјҢжө®зӮ№еһӢ
        qi::double_[ref(dTestOut) = qi::_1] >> space>>
        // з¬¬дәҢйғЁеҲҶпјҢpush_backеҮҪж•°пјҢж•ҙеһӢ
        qi::int_[push_back(boost::phoenix::ref(stIntTestOut), qi::_1)][ref(iTestOut) = qi::_1] >> space>>  
        // з¬¬дёүйғЁеҲҶпјҢиҮӘе®ҡд№үеҮҪж•°\д»ҝеҮҪж•°пјҢstlж•°жҚ®з»“жһ„ 
        (*(qi::char_))[SpiritQiStringAssign(strTestOut)];

    res = qi::parse(strTest.begin(), strTest.end(), stQiRule);
    std::cout<< "Spirit.Qi    => Line "<< std::setw(4)<< __LINE__<< ": QI::Parse complex struct "<< (res?"true": "false")<< std::endl;
    std::cout<< "Spirit.Qi    => \t\t double:"<< dTestOut<< " int:"<< stIntTestOut[0]<< " string:"<< strTestOut<< std::endl;
    

    // Qi й«ҳзә§и§„еҲҷз”ҹжҲҗеҷЁзӨәдҫӢ
    using boost::spirit::repository::confix;
    string c_commit_content, cpp_commit_content;
    auto c_comment = confix("/*", "*/")[*((char_ - "*/")[push_back(boost::phoenix::ref(c_commit_content), qi::_1)])];
    qi::rule cpp_comment = confix("//", eol)[*((char_ - eol)[push_back(boost::phoenix::ref(cpp_commit_content), qi::_1)])];

    strTest = "\r\n Hello /* This is C style comment. */ also // This is CPP style comment \r\n Oh yeah.";
    
    res = qi::parse(strTest.begin(), strTest.end(), *(*(char_ - "/*" - "//")>> (c_comment | cpp_comment)));
    std::cout<< "Spirit.Qi    => Line "<< std::setw(4)<< __LINE__<< ": QI::Parse confix rule "<< (res?"true": "false")<< std::endl;
    std::cout<< "Spirit.Qi    => \t\t c style comment:"<< c_commit_content<< std::endl;
    std::cout<< "Spirit.Qi    => \t\t c++ style comment:"<< cpp_commit_content<< std::endl;
}

иҝҷдёӘеә“жҳҜз”ЁжқҘжҠҠдёҖдәӣSTLзҡ„ж•°жҚ®з»“жһ„жҢүе’ҢQiдёҖж ·зҡ„и§„еҲҷиҪ¬еҢ–жҲҗеҲ°иҫ“еҮәжөҒзҡ„пјҢж„ҹи§үз”ЁеӨ„дёҚеӨ§пјҢеҸӘжҳҜдёҖдёӘдёәдәҶе®Ңж•ҙиҖҢеӯҳеңЁзҡ„дёңиҘҝгҖӮ еӨ§дҪ“дёҠе’ҢQiе·®дёҚеӨҡпјҢеҸӘдёҚиҝҮжҳҜеҸҚиҝҮжқҘдәҶгҖӮжҜ”еҰӮпјҢQiдҪҝз”Ёзҡ„жҳҜиҫ“е…ҘжөҒпјҢKarmaдҪҝз”Ёзҡ„жҳҜиҫ“еҮәжөҒгҖӮ еҸҰеӨ–KarmaжңүдёҖдёӘжҜ”иҫғзү№еҲ«зҡ„ең°ж–№пјҢеӣ дёәи§„еҲҷз”ҹжҲҗеӨ§еӨҡж•°зҡ„з¬¬дёҖдёӘж•°жҚ®дёҚжҳҜKarmaз»„д»¶пјҢжүҖд»ҘжңүдёӘеҮҪж•°karma::epsпјҢз”ЁдәҺз”ҹжҲҗдёҖдёӘз©әзҡ„KarmaиЎЁиҫҫејҸгҖӮ зӣҙжҺҘдёҠдҫӢзЁӢеҗ§пјҡ

// з”ҹжҲҗVector
static std::vector GenIntVec(){
    std::vector ret;
    ret.push_back(3);
    ret.push_back(1);
    ret.push_back(4);
    ret.push_back(1);
    ret.push_back(5);
    ret.push_back(9);
    ret.push_back(2);
    ret.push_back(6);
    ret.push_back(5);
    ret.push_back(3);
    ret.push_back(5);
    ret.push_back(9);

    return ret;
}

void test_spirit_karma() {
    using namespace boost::spirit;
    using boost::spirit::ascii::space;

    // ============================================================
    // ============ Spirit karma component, иҫ“еҮәз”ҹжҲҗеҷЁ ============
    // ============================================================
    std::vector stKarmaIntVec = GenIntVec();
    std::string strKarma;
    std::back_insert_iterator stStrInsertIter(strKarma);
    /**
     * еҹәжң¬и§„еҲҷзҡ„дҪҝз”Ёж–№ејҸдёә
     * karma::зұ»еһӢ            з”ҹжҲҗиҫ“еҮә. еҰӮ: karma::int_
     * karma::зұ»еһӢ(еҢ№й…ҚеҖј)    еҸӘз”ҹжҲҗеҖјдёәеҢ№й…ҚеҖјзҡ„иҫ“еҮә. karma::int_()
     * karma::eps(...) << ... << ... еӨҚеҗҲз”ҹжҲҗеҷЁ
     * karma::eps() еҮҪж•°пјҢеҪ“е…¶еҶ…йғЁsucceedеұһжҖ§иў«и®ҫдёәtrueж—¶пјҢдјҡз”ҹжҲҗиҪ¬жҚўи§„еҲҷ
     * е…·дҪ“зұ»еһӢеҸӮз…§ Boost.Spirit еә“ Karma йғЁеҲҶ Karma Generators з« иҠӮ
     *
     * karma зҡ„иҮӘе®ҡд№үи§„еҲҷгҖҒеұһжҖ§иЎҢдёәзӯүзұ»дјј qiпјҢеҸҜеҸӮз…§KarmaйғЁеҲҶзҡ„зӣёеә”з« иҠӮ
     */
    karma::generate_delimited(stStrInsertIter, karma::eps(true)<< '[' < < (karma::int_ % ',')<< ']', space, stKarmaIntVec);
    std::cout<< "Spirit.Karma => Line "<< std::setw(4)<< __LINE__<< ": Karma::generate_delimited => "<< strKarma<< std::endl;

еҸҜиғҪжңүдәәд№ӢеүҚеҗ¬иҜҙиҝҮFlexеә“пјҢз”ЁжқҘз”ҹжҲҗд»Јз Ғзҡ„гҖӮиҖҢBoost.Spiritзҡ„Lexеә“зҡ„еҫҲеӨҡең°ж–№е’Ңе®ғеҫҲеғҸпјҲжҲ‘д№ҹжІЎз”ЁиҝҮFlexпјҢе®ҳж–№жҳҜиҝҷд№ҲиҜҙзҡ„пјүгҖӮ Lexзҡ„еҘҪеӨ„е‘ўпјҢе°ұжҳҜеҸҜд»Ҙз”ЁжӯЈеҲҷиЎЁиҫҫејҸжҸҸиҝ°дёҖдёӘи§„еҲҷпјҢиҖҢдё”еҸҜд»ҘеҠЁжҖҒз”ҹжҲҗгҖӮиҖҢдё”еҸҜд»ҘеҸҜQiж··еҗҲиө·жқҘдҪҝз”ЁгҖӮ еңЁз ”з©¶иҝҷдёӘеә“зҡ„ж—¶еҖҷпјҢжҲ‘д№ҹеҗҢж—¶еҸ‘зҺ°пјҢжғіиҰҒзңҹжӯЈй«ҳж•Ҳзҡ„дҪҝз”ЁSpiritеә“пјҢиҝҳеә”иҜҘеғҸиҝҷйҮҢзҡ„дҫӢзЁӢдёҖж ·пјҢеҗ„з§ҚжЁЎжқҝз»§жүҝпјҢдҪҶжҳҜпјҢиҝҷд№ҹдјҡеўһеҠ зј–зЁӢзҡ„еӨҚжқӮеәҰгҖӮ

LexеҜ№е№¶дёҚжҳҜж”ҜжҢҒжүҖжңүжӯЈеҲҷиЎЁиҫҫејҸзҡ„иҜӯжі•пјҢе…¶ж”ҜжҢҒзҡ„жӯЈеҲҷиЎЁиҫҫејҸи§„еҲҷеҸҜд»ҘеҸӮи§Ғ Lexеә“ зҡ„ Supported Regular Expressions з« иҠӮ

еҜ№дәҺLexеә“зҡ„и§„еҲҷзұ»еһӢеҲҶзҰ»пјҢйҰ–е…ҲеҸҜд»ҘйҮҮз”Ёе’ҢFlexзұ»дјјзҡ„еҒҡжі•пјҢиҮӘе®ҡд№үж•°жҚ®еҲҶж®өеӨ„зҗҶзҡ„д»ҝеҮҪж•°пјҢеҸӘиҰҒе®ҢжҲҗ

иҝҷж ·зҡ„ж“ҚдҪңз¬ҰйҮҚиҪҪеҚіеҸҜпјҢеңЁеҮҪж•°дёӯпјҢеҸҜд»ҘйҖҡиҝҮдёҚеҗҢзҡ„IDеҢәеҲҶеҢ№й…Қзҡ„еҶ…е®№пјҢе…·дҪ“дҫӢзЁӢдёӢйқўжңү

жҲ–иҖ…пјҢе’ҢQiдёҖж ·пјҢеҸҜд»ҘдҪҝз”Ёжү©еұ•зҡ„PhoenixеҠҹиғҪе®һзҺ°з®ҖеҚ•зҡ„еҠЁдҪңеҷЁж“ҚдҪң

еҗҢж—¶пјҢLexж”ҜжҢҒе‘ҪеҗҚжЁЎејҸпјҢеҸҜд»ҘдҪҝз”Ёlex::lexer::self.add_patternжқҘеҲӣе»әе‘ҪеҗҚжЁЎејҸе’ҢдҪҝз”Ё**{еҚ дҪҚз¬ҰеҗҚз§°}жқҘи®ҫзҪ®е‘ҪеҗҚеҚ дҪҚз¬Ұзҡ„tokenе®ҡд№ү еҸҰеӨ–пјҢLexиҝҳеҸҜд»Ҙе’ҢQiз»“еҗҲдҪҝз”Ё**пјҢж— и®әжҳҜLexзҡ„жЁЎејҸз»“жһ„иҝҳжҳҜжҢүиҮӘе®ҡд№үж•°жҚ®еҲҶж®өеӨ„зҗҶд»ҝеҮҪж•°ж—¶дҪҝз”Ёзҡ„IDзј–еҸ·зҡ„ж–№жі•пјҢйғҪжңүзӣёеә”зҡ„ж–№жі•и®©д»–дҫқжҚ®Lexзҡ„и§„еҲҷеҲҶжһҗпјҢжҢүQiзҡ„еҠЁдҪңеӨ„зҗҶеҮҪж•°еӨ„зҗҶ

LexиҝҳжңүдёҖдёӘйҮҚиҰҒзҡ„йғЁеҲҶпјҢйқҷжҖҒи§„еҲҷз”ҹжҲҗгҖӮ йқҷжҖҒи§„еҲҷз”ҹжҲҗжҜ”иҫғзү№ж®ҠпјҢйҰ–е…ҲпјҢз”ҹжҲҗи§„еҲҷзҡ„еҶҷжі•е’ҢдёҠйқўжҸҗеҲ°зҡ„дёҖиҮҙпјҢдёҚеҗҢзҡ„ең°ж–№жңүдёӨзӮ№йҰ–е…ҲжҳҜlex::lexertl::lexerзҡ„дҪҝз”ЁиҰҒж”№жҲҗlex::lexertl::static_lexer з¬¬дәҢеӨ„жҜ”иҫғзү№еҲ«пјҢе°ұжҳҜйңҖиҰҒдҪҝз”Ёlex::lexertl::generate_static_dfaеҮҪж•°пјҢдҫқжҚ®и§„еҲҷпјҢз”ҹжҲҗд»Јз ҒгҖӮ

template 
struct word_count_tokens1 : lex::lexer
{
    word_count_tokens1()
    {
        this->self.add
            ("(?i:owent)", ID_OWENT)    // еҝҪз•ҘеӨ§е°ҸеҶҷпјҢз»ҹи®Ў owent дёӘж•°
            ("\n", ID_EOL)              // ж–°зҡ„дёҖиЎҢ
            (".", ID_CHAR)              // жҚўиЎҢеӨ–зҡ„д»»ж„Ҹеӯ—з¬Ұ
        ;
    }
};

struct counter1
{
    int& o;
    int& l;
    counter1(int& _o, int& _l): o(_o), l(_l){}

    template 
    bool operator()(Token const& t) const
    {
        switch (t.id()) {
        case ID_OWENT: 
            ++ o;
            break;
        case ID_EOL:
            ++l;
            break;
        case ID_CHAR:
            break;
        }

        return true;        // з»§з»ӯеҲҶжһҗ
    }
};
